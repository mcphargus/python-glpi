 48          """ 
 49          Connect to a running GLPI instance that has the webservices 
 50          plugin enabled. U{It's available here 
 51          <http://plugins.glpi-project.org/spip.php?article94>} 
 52   
 53          @type host: FQDN string 
 54          @param host: hostname of the GLPI server, has not been tested 
 55          with HTTPS 
 56          @type login_name: string 
 57          @param login_name: your GLPI username 
 58          @type login_password: string 
 59          @param login_password: pretty obvious 
 60          """ 
 61          self.url = 'http://'+host+'/plugins/webservices/rest.php?' 
 62          self.login_name = login_name 
 63          self.login_password = login_password 
 64          params = {'login_name':login_name, 
 65                    'login_password':login_password, 
 66                    'method':'glpi.doLogin'} 
 67          request = urllib2.Request(self.url + urllib.urlencode(params)) 
 68          response = urllib2.urlopen(request).read() 
 69          try: 
 70              session_id = json.loads(response)['session']        
 71              self.session = session_id 
 72          except: 
 73              raise Exception("Login incorrect or server down")
